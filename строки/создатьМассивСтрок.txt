#include <iostream>
using namespace std;
 
int main() {
        char ** strArr;
        strArr = new char*[5];
 
        for(int i=0; i<5; i++) {
                cout<<"Input string:"<<endl;
                strArr[i] = new char[255];
                cin.getline(strArr[i], 255);
        }
 
        // Print string array and free memory
        for(int i=0; i<5; i++) {
                cout<<strArr[i]<<endl;
                delete [] strArr[i];
        }
        delete [] strArr;
 system("pause");
}
